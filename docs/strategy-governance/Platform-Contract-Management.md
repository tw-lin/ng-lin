export class ContractService {
  private contractsSubject = new BehaviorSubject<Contract[]>([]);
  public contracts$ = this.contractsSubject.asObservable();

  constructor(private http: HttpClient) {}

  loadContracts(): Observable<Contract[]> {
    return this.http.get<Contract[]>('/api/contracts').pipe(
      tap(contracts => this.contractsSubject.next(contracts)),
      catchError(error => {
        console.error('è¼‰å…¥åˆç´„å¤±æ•—', error);
        return of([]);
      })
    );
  }

  parseDocument(file: File): Observable<ParsingResult> {
    const formData = new FormData();
    formData.append('file', file);
    
    return this.http.post<ParsingResult>('/api/documents/parse', formData).pipe(
      map(result => ({
        ...result,
        tasks: result.tasks.map(task => ({
          ...task,
          id: generateId()
        }))
      }))
    );
  }
}
